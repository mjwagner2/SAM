equations{ 'csp.gss.sf.total_opt_eff' } = define() {
	return ${csp.gss.sf.peak_opt_eff}*${eta_opt_soil}*${eta_opt_gen}; };

equations{ 'csp.gss.solf.fixed_land_area' } = define() {
	return ${csp.gss.sf.field_area}* 0.0002471; };

equations{ 'csp.gss.sf.peak_opt_eff' } = define() {
	maxeff = 0;
	arr = ${OpticalTable};
	iuns = ${istableunsorted};
	
	if(iuns)
	{
		for(r=0; r<#arr; r++)
		{
			if(arr[r][2] > maxeff)
				maxeff = arr[r][2];
		}
	}
	else{
	
		for (r=1;r<#arr;r++)
			for (c=1;c<#arr[0];c++)
				if (arr[r][c] > maxeff)
					maxeff = arr[r][c];
	}

	//msgbox("maxeff = " + maxeff);
	return maxeff;
};

equations{ 'qsf_des' } = define() {
	return ${w_des}
	/${eta_des}
	*${solarm};
};

equations{ 'csp.gss.solf.total_land_area' } = define() {
	return ${csp.gss.solf.fixed_land_area} * ${csp.gss.solf.land_overhead_factor}; };

equations{ 'csp.gss.sf.design_thermal_loss' } = define() {
	return ${f_sfhl_ref}*${qsf_des};
};

equations{ 'csp.gss.sf.field_area' } = define() {
	return (${qsf_des}
	+${csp.gss.sf.design_thermal_loss})
	/${csp.gss.sf.total_opt_eff}
	*1000000
	/${irr_des};
};

equations{ 'track_mode' } = define() {
	if ( 1==${csp.gss.sf.rad_type} )
		return 0;
	else
		return 1; // TRNSYS code sets this to 3
};

equations{ 'rad_type' } = define() {
	return ${csp.gss.sf.rad_type}+1;
};

equations{ 'sfhlQ_coefs' } = define() {
	arr[3] = ${csp.gss.sf.irr_loss_f3};

	arr[0] = ${csp.gss.sf.irr_loss_f0};
	return arr;

on_load{ "Generic CSP Solar Field" } = define()
{
	on_change{"istableunsorted"}();
};



on_change{ "istableunsorted" } = define()
{
	iss = value("istableunsorted");
	
	iss0 = value("istableunsorted0");	//last stored state
	
	//if unsorted, change the properties of the table
	if( iss )
	{
		property("OpticalTable", "ShadeR0C0", false);
		property("OpticalTable", "ShowColLabels", true);
		property("OpticalTable", "ColLabels", ["Azimuth", "Zenith", "Efficiency"]);
		value("interp_arr", false);
		enable("interp_arr", false);
		property("Label1", "Caption", "Notes: Optical performance is defined in a 3-column table using a list of optical efficiencies at various azimuth/zenith angle combinations. Each row defines an azimuth / zenith / efficiency data point.");
	}
	else
	{
		property("OpticalTable", "ShadeR0C0", true);
		property("OpticalTable", "ShowColLabels", false);
		enable("interp_arr", true);
		value("interp_arr", true);
		property("Label1", "Caption", "Notes: Optical performance is defined in a table of efficiencies at various azimuth/zenith angles.  The first row in the table specifies the azimuth angles, and the first column specifies the zenith angle.  The first cell (0,0) is unused.  Angles are specified in degrees, and the corresponding efficiencies in fraction (0...1).");
	}
	
	// swap tables
	if(iss != iss0)
	{
		tab = ${OpticalTable};
		value("OpticalTable", ${OpticalTable_D});	
		value("OpticalTable_D", tab);
	}
	
	// update saved last state
	value("istableunsorted0", iss);	
};
