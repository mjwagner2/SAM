	value("T_htf_cold_calc",nan());

function sco2_design_point() {
	obj = ssc_create();
	ssc_var(obj, "W_dot_net_des",${P_ref});
	ssc_var(obj, "eta_c",${eta_c});
	ssc_var(obj, "eta_t",${eta_t});
	ssc_var(obj, "P_high_limit",${P_high_limit});
	ssc_var(obj, "deltaT_PHX",${deltaT_PHX});
	ssc_var(obj, "deltaT_ACC",${T_ITD_des});
	ssc_var(obj, "T_amb_des",${T_amb_des});
	ssc_var(obj, "T_htf_hot_des",${T_htf_hot_des});
	ssc_var(obj, "eta_des",${design_eff});
	ssc_var(obj, "run_off_des_study",0);
	
	logmsg("sCO2 Design Point Optimization");
	
	ret = ssc_exec(obj, "sco2_design_point", {'show_dialog'=true, 'hold_dialog'=true});
	
	if(ret != 0)
	{	
		ssc_free( obj );
		zero_sco2_design_point();
		return;
	}
	
	// If successful, update flag
	value("is_sco2_optimized",1.0);
	// And hide optimization button
	show("btn_sco2_design_point", false);
	// And show label
	show("optimized_des_point_pars", ${pc_config}==1);
	// refresh
	refresh();
	
	UA_total = ssc_var(obj, "UA_total");
	recomp_frac = ssc_var(obj, "recomp_frac");
	P_comp_in = ssc_var(obj, "P_comp_in");
	P_comp_out = ssc_var(obj, "P_comp_out");
	T_htf_cold = ssc_var(obj, "T_htf_cold");
	
	value("T_htf_cold_calc",T_htf_cold);
	value("UA_total_calc",UA_total);
	value("recomp_frac_calc",recomp_frac);
	value("P_in_calc",P_comp_in);
	value("P_out_calc",P_comp_out);
	
	ssc_free( obj );	
}
