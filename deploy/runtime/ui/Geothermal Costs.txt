function reset_geothermal_plant_cost()
{
	// This should never get called for flash plants - the checkbox that calls this code is disabled for flash plants
	if (value("conversion_type") == 0) // binary plant
	{
        if ( ( value("geotherm.cost.plant_auto_estimate") == 0 ) && ( value("geotherm.cost.plant_per_kW.bak") != 0 ) )
		{
            // The users original value will get saved into "geotherm.cost.plant_per_kW.bak" the first time they
            // check the box to automatically calculate costs.  This is done in mimo_geothermal_eval, in samwx\samexe\src\uievalspecial.cpp

			// When they uncheck the box, put back user's original value, and set the backup = 0 (otherwise it won't be written
            // over again when they re-check the box
            value("geotherm.cost.plant_per_kW", value("geotherm.cost.plant_per_kW.bak") );
            value("geotherm.cost.plant_per_kW.bak", 0 );

		}
	}
}

function update_cost_input_control()
{
	enable("geotherm.cost.plant_per_kW", value("geotherm.cost.plant_auto_estimate") == 0 );
}


on_change{"geotherm.cost.plant_auto_estimate"} = define()
{
	update_cost_input_control();
	reset_geothermal_plant_cost();
};

on_load{'Geothermal Costs'} = define() {
	if (value("conversion_type") == 0) // binary plant
	{
		value("geotherm.cost.plant_auto_estimate", value("geotherm.cost.plant_auto_estimate_last_setting") );
	}
	else // this is a flash plant
	{
		value("geotherm.cost.plant_auto_estimate_last_setting", value("geotherm.cost.plant_auto_estimate")); // save users choice for binary
		value("geotherm.cost.plant_auto_estimate", 0);
	}
	
	// if this is a binary plant, enable; otherwise, disable
	enable("geotherm.cost.plant_auto_estimate", (${conversion_type} == 0) );

	update_cost_input_control();
};
