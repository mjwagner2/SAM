equations{ 'csp_dtr_sca_ap_length_1' } = define() {
	return ${csp_dtr_sca_length_1}/${csp_dtr_sca_ncol_per_sca_1};

equations{ 'csp_dtr_sca_calc_end_gain_1' } = define() {
	return  max(${csp_dtr_sca_ave_focal_len_1} 
				* tan( ${csp_dtr_sca_calc_theta_1} ) 
				- ${csp_dtr_sca_piping_dist_1}, 0);
};

equations{ 'csp_dtr_sca_calc_costh_1' } = define() {
	return  sqrt(1 - pow(cos(1.57-${csp_dtr_sca_calc_zenith_1}-${tilt}) 
             - cos(${tilt}) 
               * cos(1.57-${csp_dtr_sca_calc_zenith_1}) 
               * (1 - cos(0 - ${azimuth})), 2) 
			);
};

equations{ 'csp_dtr_sca_calc_end_loss_1' } = define() {
	return  1 - (${csp_dtr_sca_ave_focal_len_1}*tan(${csp_dtr_sca_calc_theta_1}) 
- ( ${nSCA} - 1) / ${nSCA} * ${csp_dtr_sca_calc_end_gain_1})
 / (${csp_dtr_sca_length_1}*${csp_dtr_sca_ncol_per_sca_1}); };

equations{ 'csp_dtr_sca_calc_sca_eff_1' } = define() {
	return  ${csp_dtr_sca_tracking_error_1}* ${csp_dtr_sca_geometry_effects_1}
			* ${csp_dtr_sca_clean_reflectivity_1} * ${csp_dtr_sca_mirror_dirt_1}
			* ${csp_dtr_sca_general_error_1}; };

equations{ 'csp_dtr_sca_calc_latitude_1' } = define() {
	return ${lat}; };

equations{ 'csp_dtr_sca_calc_zenith_1' } = define() {
	return pi()/180*(90-(90-(${lat}-23.5))); };

			IAM = ${IAMs_1}[0];
			l_IAM = #${IAMs_1};

				for(i=1;i<l_IAM;i++)
